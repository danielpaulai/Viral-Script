async function callOpenAI(prompt) {
  // Pre-flight check
  if (!process.env.OPENAI_API_KEY) {
    console.error('OpenAI API call attempted but OPENAI_API_KEY is not set');
    throw new Error('Server configuration error: OpenAI API key is not available. Please contact support.');
  }

  try {
    const response = await openai.chat.completions.create({
      model: "gpt-4",
      messages: [{ role: "user", content: prompt }],
    });
    
    return response.choices[0].message.content;
    
  } catch (error) {
    console.error('OpenAI API Error:', error.message);
    
    // Specific error handling
    if (error.message.includes('API key')) {
      console.error('Authentication error - API key may be invalid');
      throw new Error('Authentication failed. Please check API key configuration.');

    if (error.message.includes('rate limit')) {
      console.error('Rate limit exceeded');
      throw new Error('Service temporarily unavailable. Please try again in a moment.');
    }
    
    if (error.message.includes('insufficient_quota')) {
      console.error('OpenAI quota exceeded');
      throw new Error('API quota exceeded. Please check your OpenAI billing.');
    }
    
    throw error;
  }
}
